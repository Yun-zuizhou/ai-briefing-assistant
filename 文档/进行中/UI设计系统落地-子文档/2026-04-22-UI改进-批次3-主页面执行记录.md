#### 修改项2：meta/metaLinks改为可选
```tsx
interface MastheadProps {
  title: string;
  subtitle?: string;
  showMeta?: boolean;
  meta?: ReactNode;
  metaLinks?: ReactNode;
}

// 默认不显示meta
export function Masthead({ title, subtitle, showMeta = false, meta, metaLinks }: MastheadProps) {
  return (
    <div className="masthead">
      <h1 className="masthead-title">{title}</h1>
      {subtitle && <p className="masthead-subtitle">{subtitle}</p>}
      {showMeta && meta && <div className="masthead-meta">{meta}</div>}
      {showMeta && metaLinks && <div className="masthead-meta-links">{metaLinks}</div>}
    </div>
  );
}
```
