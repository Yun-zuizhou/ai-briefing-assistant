```tsx
// 原内联组件提取到 components/ui/StatusPill.tsx
interface StatusPillProps {
  status: 'idle' | 'loading' | 'success' | 'error';
  children: ReactNode;
}

export default function StatusPill({ status, children }: StatusPillProps) {
  const statusStyles = {
    idle: 'bg-paper-warm text-ink-muted',
    loading: 'bg-paper-warm text-accent',
    success: 'bg-paper-warm text-accent',
    error: 'bg-paper-warm text-red-600',
  };

  return (
    <span className={`inline-flex items-center gap-1 px-2 py-1 text-xs border ${statusStyles[status]}`}>
      {status === 'loading' && <LoadingSpinner size={12} />}
      {children}
    </span>
  );
}
```
