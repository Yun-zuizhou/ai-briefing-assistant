## 2. 文件修改清单

### 文件1：`apps/web/src/components/ui/ReportLayout.tsx`（新增）

```tsx
import type { ReactNode } from 'react';
import { Download, Share2 } from 'lucide-react';
import { IconButton } from './index';

interface ReportLayoutProps {
  title: string;
  period: string;
  children: ReactNode;
  onExport?: () => void;
  onShare?: () => void;
}

export default function ReportLayout({ title, period, children, onExport, onShare }: ReportLayoutProps) {
  return (
    <div className="report-layout">
      <div className="report-header">
        <div>
          <h1 className="report-title">{title}</h1>
          <p className="report-period">{period}</p>
        </div>
        <div className="flex gap-2">
          {onExport && (
            <IconButton
              icon={Download}
              ariaLabel="导出报告"
              onClick={onExport}
            />
          )}
          {onShare && (
            <IconButton
              icon={Share2}
              ariaLabel="分享报告"
              onClick={onShare}
            />
          )}
        </div>
      </div>
      <div className="report-content">
        {children}
      </div>
    </div>
  );
}
```
