### 1.2 技术选型总览

| 层级 | 技术选型 | 备选方案 | 选择理由 |
|-----|---------|---------|---------|
| **移动端** | React Native | Flutter | 跨平台、生态成熟、学习曲线平缓 |
| **Web管理端** | React + Ant Design | Vue + Element UI | 生态丰富、组件完善 |
| **后端框架** | FastAPI (Python) | Express (Node.js) | 性能优秀、异步支持、AI生态 |
| **数据库** | MySQL 8.0 | PostgreSQL | 成熟稳定、文档丰富 |
| **缓存** | Redis | Memcached | 功能丰富、支持持久化 |
| **搜索引擎** | Elasticsearch | MeiliSearch | 全文搜索、高性能 |
| **爬虫框架** | Scrapy | requests + BeautifulSoup | 功能完整、可扩展 |
| **消息队列** | Celery + Redis | RabbitMQ | Python生态、简单易用 |
| **部署** | Docker + Nginx | K8s | 简单可靠、资源占用低 |

#### 核心表结构

```sql
CREATE TABLE users (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    avatar VARCHAR(255),
    status TINYINT DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_email (email),
    INDEX idx_username (username)
);

CREATE TABLE sources (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    url VARCHAR(500) NOT NULL,
    type ENUM('zhenggao', 'jianzhi', 'ai_news', 'weekly_news', 'custom') NOT NULL,
    credibility DECIMAL(3,2) DEFAULT 0.50,
    status ENUM('active', 'inactive', 'error') DEFAULT 'active',
    config JSON,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_type (type),
    INDEX idx_status (status)
);

CREATE TABLE infos (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(500) NOT NULL,
    content TEXT,
    summary VARCHAR(1000),
    source_id BIGINT,
    category VARCHAR(50),
    tags JSON,
    publish_time TIMESTAMP,
    original_url VARCHAR(500),
    credibility DECIMAL(3,2),
    view_count INT DEFAULT 0,
    collect_count INT DEFAULT 0,
    status ENUM('active', 'deleted') DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (source_id) REFERENCES sources(id),
    INDEX idx_category (category),
    INDEX idx_publish_time (publish_time),
    INDEX idx_source (source_id),
    FULLTEXT INDEX ft_title_content (title, content)
);

CREATE TABLE user_feedbacks (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    info_id BIGINT NOT NULL,
    type ENUM('useful', 'useless', 'collect', 'report') NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (info_id) REFERENCES infos(id),
    UNIQUE KEY uk_user_info_type (user_id, info_id, type),
    INDEX idx_user (user_id),
    INDEX idx_info (info_id)
);

CREATE TABLE user_ideas (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    info_id BIGINT,
    content TEXT NOT NULL,
    tags JSON,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (info_id) REFERENCES infos(id),
    INDEX idx_user (user_id),
    INDEX idx_created (created_at)
);

CREATE TABLE user_profiles (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT UNIQUE NOT NULL,
    interests JSON,
    behavior_stats JSON,
    reading_habits JSON,
    portrait TEXT,
    generated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE user_subscripts (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    keyword VARCHAR(100),
    category VARCHAR(50),
    frequency ENUM('realtime', 'daily', 'weekly') DEFAULT 'daily',
    enabled BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    INDEX idx_user (user_id),
    INDEX idx_enabled (enabled)
);

CREATE TABLE crawl_tasks (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    source_id BIGINT NOT NULL,
    status ENUM('pending', 'running', 'success', 'failed') DEFAULT 'pending',
    last_run TIMESTAMP,
    next_run TIMESTAMP,
    error_count INT DEFAULT 0,
    last_error TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (source_id) REFERENCES sources(id),
    INDEX idx_status (status),
    INDEX idx_next_run (next_run)
);
```

## 五、AI模块设计

### 5.1 意图识别模块

```python
from typing import Dict, List, Optional
from dataclasses import dataclass

@dataclass
class Intent:
    name: str
    confidence: float
    slots: Dict[str, str]

class IntentRecognizer:
    def __init__(self):
        self.intent_patterns = {
            'zhenggao': ['征稿', '投稿', '稿费', '写作'],
            'jianzhi': ['兼职', '副业', '远程工作', '外包'],
            'ai_news': ['AI', '人工智能', '机器学习', '大模型'],
            'weekly_news': ['新闻', '周报', '热点', '时事'],
        }
    
    def recognize(self, text: str) -> Intent:
        text_lower = text.lower()
        scores = {}
        
        for intent, keywords in self.intent_patterns.items():
            score = sum(1 for kw in keywords if kw in text_lower)
            scores[intent] = score
        
        if not scores or max(scores.values()) == 0:
            return Intent(name='unknown', confidence=0.0, slots={})
        
        best_intent = max(scores, key=scores.get)
        confidence = scores[best_intent] / len(self.intent_patterns[best_intent])
        
        slots = self._extract_slots(text, best_intent)
        
        return Intent(
            name=best_intent,
            confidence=confidence,
            slots=slots
        )
    
    def _extract_slots(self, text: str, intent: str) -> Dict[str, str]:
        slots = {}
        if intent == 'zhenggao':
            if '科技' in text:
                slots['category'] = 'tech'
            if '文学' in text:
                slots['category'] = 'literature'
        return slots
```

### 5.2 用户画像生成

```python
from typing import List, Dict
from collections import Counter
from datetime import datetime, timedelta

class ProfileGenerator:
    def generate(self, user_id: int) -> Dict:
        feedbacks = self._get_user_feedbacks(user_id)
        ideas = self._get_user_ideas(user_id)
        behaviors = self._get_user_behaviors(user_id)
        
        interests = self._analyze_interests(feedbacks, ideas)
        habits = self._analyze_habits(behaviors)
        stats = self._calculate_stats(feedbacks, behaviors)
        portrait = self._generate_portrait(interests, habits, stats)
        
        return {
            'interests': interests,
            'habits': habits,
            'stats': stats,
            'portrait': portrait,
            'generated_at': datetime.now()
        }
    
    def _analyze_interests(self, feedbacks, ideas) -> List[Dict]:
        categories = [f.info.category for f in feedbacks if f.type == 'useful']
        tags = []
        for idea in ideas:
            tags.extend(idea.tags or [])
        
        category_counter = Counter(categories)
        tag_counter = Counter(tags)
        
        interests = []
        for cat, count in category_counter.most_common(10):
            interests.append({
                'name': cat,
                'weight': count / len(categories),
                'type': 'category'
            })
        
        for tag, count in tag_counter.most_common(10):
            interests.append({
                'name': tag,
                'weight': count / len(tags),
                'type': 'tag'
            })
        
        return interests
    
    def _generate_portrait(self, interests, habits, stats) -> str:
        top_interests = [i['name'] for i in interests[:3] if i['type'] == 'category']
        
        portrait = f"你是一位对{', '.join(top_interests)}等领域充满热情的用户。"
        portrait += f"在过去30天里，你阅读了{stats['total_read']}篇文章，"
        portrait += f"平均每篇停留{stats['avg_read_time']:.1f}分钟。"
        
        if habits['peak_hours']:
            hours = habits['peak_hours']
            portrait += f"你倾向于在{hours[0]}-{hours[1]}点阅读信息。"
        
        return portrait
```
